My final version after modifications:

```javascript
const scores = [85, 92, 78, 88, 95];

// Calculate the average of all numbers in an array
function getAverage(arr) {
  let sum = 0;
  for (const n of arr) {
    sum += n;
  }
  return sum / arr.length;
}

// Return the largest number in an array
function getHighest(arr) {
  let highest = arr[0];
  for (const n of arr) {
    if (n > highest) {
      highest = n;
    }
  }
  return highest;
}

console.log("Average:", getAverage(scores));   // 87.6
console.log("Highest:", getHighest(scores));   // 95
```

Verified the output matches the original messy code: average is 87.6, highest is 95.
